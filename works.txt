Ito yung ippaste niyo dun sa SQL localhost.

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    gender VARCHAR(50),
    password VARCHAR(255) NOT NULL
);

CREATE TABLE issues (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tracking_id VARCHAR(50) UNIQUE,
    user VARCHAR(255),
    title VARCHAR(255),
    description TEXT,
    category VARCHAR(100),
    barangay VARCHAR(100),
    street VARCHAR(100),
    status VARCHAR(50) DEFAULT 'Pending',
    image_path VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE issues 
ADD barangay VARCHAR(100) AFTER category,
ADD street VARCHAR(100) AFTER barangay;

CREATE TABLE messages (
    id INT AUTO_INCREMENT PRIMARY KEY,
    issue_tracking_id VARCHAR(50),
    sender VARCHAR(255),
    text TEXT,
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (issue_tracking_id) REFERENCES issues(tracking_id)
);
